34int gdhmap_int_prime(int n) {
35 //nearest odd integer that is less than n
36 int odd_m = (n % 2 == 0) ? n-1 : n-2;

102int gdhmap_put(gdhmap* gdhmap, char* key, void* value, size_t valuesize) {
103 //build hash and offsets for probing
104 int hash = gdhmap_hash1(gdhmap->max_size, key);
105 int offset= gdhmap_hash2(gdhmap->__prime, key);
106
107 for(size_t i = 0; i < gdhmap->max_size; i++) {
108 if(gdhmap->keys[hash] == NULL || gdhmap->keys[hash] == DEL_MARKER) {
109 //insert if empty
110 // We want to store keys and values ourselves and not rely on pointers to variables that may get freed later
111 // -> malloc and copy
112 gdhmap->keys[hash] = malloc(strlen(key));
113 gdhmap->keys[hash] = strcpy(gdhmap->keys[hash], key);
114 gdhmap->values[hash] = malloc(valuesize);
115 memcpy(gdhmap->values[hash], value, valuesize);
116 gdhmap->size++;
117 return 1;
118 } else if(strcmp(gdhmap->keys[hash], key) == 0) {
119 //key already exists
120 return 0;
121 }
122
123 hash = gdhmap_nhash1(gdhmap->max_size, hash + offset);
124 }

126 return 0;
